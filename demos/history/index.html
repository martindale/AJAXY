Example of binding overloading: 

var $state = $('#state'),
	$custom = $('#custom'),
	$overloading = $('#overloading');

$.History.bind(function(state){
	// Our Generic Handler - Always runs last.
	$state.text(state);
});

$.History.bind('two',function(state){
	// Custom Handler for: two
	$custom.text('two is one more than one!');
});

$.History.bind('three',function(state){
	// Custom Handler for: three
	$overloading.text('three is 1');
});
$.History.bind('three',function(state){
	// Custom Handler for: three
	$overloading.text($overloading.text()+'+2');
});
$.History.bind('three',function(state){
	// Custom Handler for: three
	$overloading.text($overloading.text()+'=3');
});

$.History.bind('reset',function(state){
	// Custom Handler for: reset
	$state.add($custom).add($overloading).text('');
});
